
# cria o ambiente virtual python
python -m venv venv

# inicia a bagaça no mac
source venv/bin/activate

# requirements.txt para o Client MCP
mcp>=0.9.0
groq>=0.8.0
python-dotenv>=1.0.0

# .env - Configuração de variáveis
GROQ_API_KEY=gsk_your_groq_api_key_here
BRAVE_API_KEY=your_brave_api_key_here

# Instala dependências Python
echo "📦 Instalando dependências Python..."
pip install mcp groq python-dotenv

# Instala servidores MCP via NPM
echo "🔧 Instalando servidores MCP..."
npm install -g @modelcontextprotocol/server-filesystem
npm install -g @modelcontextprotocol/server-brave-search

# Cria arquivo .env se não existir
if [ ! -f .env ]; then
    echo "📝 Criando arquivo .env..."
    cat > .env << 'EOF'
# Chave da API do Groq (obtenha em: https://console.groq.com/keys)
GROQ_API_KEY=gsk_your_groq_api_key_here

# Chave da API Brave Search (obtenha em: https://api.search.brave.com/app/keys)
BRAVE_API_KEY=your_brave_api_key_here
EOF
    echo "⚠️  Edite o arquivo .env com suas chaves de API!"
fi

# Rodando os servidores MCP
npx -y @modelcontextprotocol/server-filesystem /tmp
BRAVE_API_KEY="sua_chave" npx -y @modelcontextprotocol/server-brave-search
